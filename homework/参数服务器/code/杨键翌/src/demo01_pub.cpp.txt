#include "ros/ros.h"
#include "std_msgs/String.h"

int main(int argc, char *argv[])
{
    setlocale(LC_ALL,"");
    //初始化节点
    ros::init(argc,argv,"yaml_publisher");
    //创建节点
    ros::NodeHandle nh;
    //创建发布者对象
    ros::Publisher pub = nh.advertise<std_msgs::String>("yaml_param",10);//不超过10条缓存
    //创建被发布的消息
    std_msgs::String my_param;
    if (!nh.getParam("params", my_param.data)) {
    ROS_ERROR("Failed to get param 'my_param_name'");
    return 1;
    }

    ROS_INFO("My param is: %s", my_param.data.c_str());
    //10Hz
    ros::Rate rate(1);
    //填入发布内容
 

    while(ros::ok()){
     
        pub.publish(my_param);
        //调试
        ROS_INFO("发布的数据是：%s",my_param.data.c_str());
        rate.sleep();
        ros::Duration(3).sleep();
        ros::spinOnce();
    }

    return 0;
}
